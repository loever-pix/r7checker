# R6 Skin Locker Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a local Node.js/Express web app where users log in with their Ubisoft credentials to view their full Rainbow Six Siege skin locker.

**Architecture:** Express backend handles Ubisoft API auth and data fetching (bypassing browser CORS restrictions); frontend is plain HTML/CSS/JS that renders the returned JSON. Player data held in `sessionStorage` — nothing persisted to disk.

**Tech Stack:** Node.js, Express, axios, HTML5, CSS3, vanilla JavaScript

### Task 2: Ubisoft Auth Module

**Files:**
- Create: `lib/auth.js`

- [ ] **Step 1: Create `lib/auth.js`**

```js
const axios = require('axios');

const UBI_APP_ID = '39baebad-39e5-4552-8c25-2c9b919064e2';
const AUTH_URL = 'https://public-ubiservices.ubi.com/v3/profiles/sessions';

const BASE_HEADERS = {
  'Content-Type': 'application/json',
  'Ubi-AppId': UBI_APP_ID,
  'Ubi-RequestedPlatformType': 'uplay',
  'User-Agent': 'UbiServices_SDK_2019.Release.58_PC64_ansi_static',
};

async function login(email, password) {
  const credentials = Buffer.from(`${email}:${password}`).toString('base64');
  const { data } = await axios.post(AUTH_URL, {}, {
    headers: { ...BASE_HEADERS, Authorization: `Basic ${credentials}` },
  });

  if (data.twoFactorAuthenticationTicket) {
    return { requires2FA: true, twoFATicket: data.twoFactorAuthenticationTicket };
  }

  return {
    requires2FA: false,
    ticket: data.ticket,
    sessionId: data.sessionId,
    userId: data.userId,
    nameOnPlatform: data.nameOnPlatform,
  };
}

async function verify2FA(code, twoFATicket) {
  const { data } = await axios.post(
    AUTH_URL,
    { code, method: 'email', trustDevice: false },
    {
      headers: {
        ...BASE_HEADERS,
        Authorization: `ubi_2fa_v1 t=${twoFATicket}`,
      },
    }
  );

  return {
    ticket: data.ticket,
    sessionId: data.sessionId,
    userId: data.userId,
    nameOnPlatform: data.nameOnPlatform,
  };
}

module.exports = { login, verify2FA, BASE_HEADERS, UBI_APP_ID };
```

const axios = require('axios');
const { BASE_HEADERS, UBI_APP_ID } = require('./auth');

const PC_SPACE_ID = '5172a557-50b5-4665-b7db-e3f2d8fbe139';
const PC_SANDBOX_ID = 'OSBOR_PC_LNCH_N_RTAL';

function authHeaders(ticket, sessionId) {
  return {
    ...BASE_HEADERS,
    Authorization: `Ubi_v1 t=${ticket}`,
    'Ubi-SessionId': sessionId,
  };
}

const RANK_NAMES = [
  'Unranked',
  'Copper V','Copper IV','Copper III','Copper II','Copper I',
  'Bronze V','Bronze IV','Bronze III','Bronze II','Bronze I',
  'Silver V','Silver IV','Silver III','Silver II','Silver I',
  'Gold V','Gold IV','Gold III','Gold II','Gold I',
  'Platinum V','Platinum IV','Platinum III','Platinum II','Platinum I',
  'Emerald V','Emerald IV','Emerald III','Emerald II','Emerald I',
  'Diamond V','Diamond IV','Diamond III','Diamond II','Diamond I',
  'Champion',
];

function getRankName(rank) {
  return RANK_NAMES[rank] ?? 'Unranked';
}

function getRankTier(rankName) {
  if (rankName.startsWith('Champion')) return 'champion';
  if (rankName.startsWith('Diamond')) return 'diamond';
  if (rankName.startsWith('Emerald')) return 'emerald';
  if (rankName.startsWith('Platinum')) return 'platinum';
  if (rankName.startsWith('Gold')) return 'gold';
  if (rankName.startsWith('Silver')) return 'silver';
  if (rankName.startsWith('Bronze')) return 'bronze';
  if (rankName.startsWith('Copper')) return 'copper';
  return 'unranked';
}

async function getProfile(userId, ticket, sessionId) {
  const headers = authHeaders(ticket, sessionId);

  const [profileRes, progressRes, linksRes] = await Promise.all([
    axios.get(`https://public-ubiservices.ubi.com/v2/profiles?userId=${userId}`, { headers }),
    axios.get(
      `https://public-ubiservices.ubi.com/v1/spaces/${PC_SPACE_ID}/sandboxes/${PC_SANDBOX_ID}/r6playerprofile/playerprofile/progressions?profile_ids=${userId}`,
      { headers }
    ),
    axios.get(`https://public-ubiservices.ubi.com/v3/users/${userId}/profiles`, { headers }),
  ]);

  const profile = profileRes.data.profiles?.[0] ?? {};
  const progress = progressRes.data.player_profiles?.[0] ?? {};
  const linkedPlatforms = (linksRes.data.profiles ?? []).map(p => p.platformType);

  return {
    username: profile.nameOnPlatform ?? '',
    avatar: `https://ubisoft-avatars.akamaized.net/${userId}/default_146_146.png`,
    level: progress.level ?? 0,
    renown: progress.xp ?? 0,
    credits: 0,
    linkedPlatforms,
  };
}

const SEASON_NAMES = {
  1: 'Black Ice', 2: 'Dust Line', 3: 'Skull Rain', 4: 'Red Crow',
  5: 'Velvet Shell', 6: 'Health', 7: 'Blood Orchid', 8: 'White Noise',
  9: 'Chimera', 10: 'Para Bellum', 11: 'Grim Sky', 12: 'Wind Bastion',
  13: 'Burnt Horizon', 14: 'Phantom Sight', 15: 'Ember Rise', 16: 'Shifting Tides',

  29: 'Commanding Force', 30: 'Dread Factor', 31: 'Heavy Mettle', 32: 'Deep Freeze',
  33: 'Deadly Omen', 34: 'New Blood', 35: 'Prep Phase', 36: 'Tenfold Pursuit',
};

async function getSeasonRanks(userId, ticket, sessionId) {
  const headers = authHeaders(ticket, sessionId);
  const CURRENT_SEASON = 36;
  const SEASONS_TO_FETCH = 15;
  const ranks = [];

  const promises = [];
  for (let s = CURRENT_SEASON; s > CURRENT_SEASON - SEASONS_TO_FETCH; s--) {
    const season = s;
    promises.push(
      axios.get(
        `https://public-ubiservices.ubi.com/v2/spaces/${PC_SPACE_ID}/sandboxes/${PC_SANDBOX_ID}/r6karma/players?board_id=pvp_ranked&profile_ids=${userId}&region_id=ncsa&season_id=${season}`,
        { headers }
      ).then(res => ({ season, data: res.data })).catch(() => null)
    );
  }

  const results = await Promise.all(promises);

  for (const result of results) {
    if (!result) continue;
    const player = result.data?.players?.[userId];
    if (!player || player.rank === 0) continue;
    const rankName = getRankName(player.rank);
    ranks.push({
      season: result.season,
      seasonName: SEASON_NAMES[result.season] ?? `Season ${result.season}`,
      rank: player.rank,
      rankName,
      rankTier: getRankTier(rankName),
      mmr: player.mmr ?? 0,
    });
  }

  return ranks;
}

function extractRarity(tags) {
  for (const t of tags) {
    if (typeof t !== 'string') continue;
    const lower = t.toLowerCase();
    if (lower.includes('legendary')) return 'Legendary';
    if (lower.includes('epic')) return 'Epic';
    if (lower.includes('rare')) return 'Rare';
    if (lower.includes('uncommon')) return 'Uncommon';
    if (lower.includes('common')) return 'Common';
  }
  return 'Standard';
}

function extractType(tags) {
  for (const t of tags) {
    if (typeof t !== 'string') continue;
    if (t.includes('CharmAttachment')) return 'Charm';
    if (t.includes('WeaponSkin')) return 'Weapon Skin';
    if (t.includes('Headgear')) return 'Headgear';
    if (t.includes('Uniform')) return 'Uniform';
    if (t.includes('Attachment')) return 'Attachment';
  }
  return 'Item';
}

function normalizeItem(i) {
  const tags = i.Tags ?? i.tags ?? [];
  return {
    id: i.ItemId ?? i.itemId ?? '',
    name: i.ItemName ?? i.itemName ?? i.ItemId ?? 'Unknown',
    image: i.ImageUrl ?? i.imageUrl ?? null,
    rarity: extractRarity(tags),
    type: extractType(tags),
  };
}

async function getInventory(userId, ticket, sessionId) {
  const headers = authHeaders(ticket, sessionId);
  let items = [];

  try {
    const res = await axios.get(
      `https://public-ubiservices.ubi.com/v1/profiles/${userId}/items?spaceId=${PC_SPACE_ID}&sandboxId=${PC_SANDBOX_ID}`,
      { headers }
    );
    items = res.data.ItemInstances ?? res.data.items ?? [];
  } catch (e) {
    console.warn('Inventory fetch failed:', e.message);
    return { seasonals: [], universals: [], blackIces: [], rankedCharms: [], attachmentSkins: [] };
  }

  function byTag(tag) {
    return items
      .filter(i => {
        const tags = (i.Tags ?? i.tags ?? []).map(t => (typeof t === 'string' ? t.toLowerCase() : ''));
        return tags.some(t => t.includes(tag.toLowerCase()));
      })
      .map(normalizeItem);
  }

  return {
    seasonals: byTag('seasonal'),
    universals: byTag('universal'),
    blackIces: byTag('blackice'),
    rankedCharms: byTag('rankedcharm'),
    attachmentSkins: byTag('weaponskin'),
  };
}

async function getPlayerData(userId, ticket, sessionId) {
  const [profile, seasonRanks, inventory] = await Promise.all([
    getProfile(userId, ticket, sessionId),
    getSeasonRanks(userId, ticket, sessionId),
    getInventory(userId, ticket, sessionId),
  ]);
  return { ...profile, seasonRanks, ...inventory };
}

module.exports = { getPlayerData };

### Task 4: Express Server

**Files:**
- Create: `server.js`

- [ ] **Step 1: Create `server.js`**

```js
const express = require('express');
const path = require('path');
const { login, verify2FA } = require('./lib/auth');
const { getPlayerData } = require('./lib/player');

const app = express();
app.use(express.json());
app.use(express.static(path.join(__dirname, 'public')));

// In-memory pending 2FA tickets: ticketId -> { twoFATicket, expiresAt }
const pendingTickets = new Map();

setInterval(() => {
  const now = Date.now();
  for (const [key, val] of pendingTickets.entries()) {
    if (val.expiresAt < now) pendingTickets.delete(key);
  }
}, 5 * 60 * 1000);

app.post('/api/login', async (req, res) => {
  const { email, password } = req.body ?? {};
  if (!email || !password) return res.status(400).json({ error: 'Email and password required' });

  try {
    const result = await login(email, password);

    if (result.requires2FA) {
      const ticketId = Math.random().toString(36).slice(2) + Date.now().toString(36);
      pendingTickets.set(ticketId, {
        twoFATicket: result.twoFATicket,
        expiresAt: Date.now() + 10 * 60 * 1000,
      });
      return res.json({ requires2FA: true, ticketId });
    }

    const playerData = await getPlayerData(result.userId, result.ticket, result.sessionId);
    res.json({ requires2FA: false, playerData });
  } catch (err) {
    const status = err.response?.status;
    if (status === 401) return res.status(401).json({ error: 'Invalid email or password' });
    console.error('Login error:', err.message);
    res.status(500).json({ error: 'Could not reach Ubisoft servers. Try again.' });
  }
});

app.post('/api/verify-2fa', async (req, res) => {
  const { code, ticketId } = req.body ?? {};
  if (!code || !ticketId) return res.status(400).json({ error: 'Code and ticketId required' });

  const pending = pendingTickets.get(ticketId);
  if (!pending || pending.expiresAt < Date.now()) {
    pendingTickets.delete(ticketId);
    return res.status(400).json({ error: 'Session expired. Please log in again.' });
  }

  try {
    const result = await verify2FA(code, pending.twoFATicket);
    pendingTickets.delete(ticketId);
    const playerData = await getPlayerData(result.userId, result.ticket, result.sessionId);
    res.json({ playerData });
  } catch (err) {
    const status = err.response?.status;
    if (status === 401) return res.status(401).json({ error: 'Invalid 2FA code' });
    console.error('2FA error:', err.message);
    res.status(500).json({ error: '2FA verification failed. Try again.' });
  }
});

const PORT = 3000;
app.listen(PORT, () => console.log(`R6 Locker running at http://localhost:${PORT}`));
```

// ── Shared helpers ────────────────────────────────

function qs(sel, root = document) { return root.querySelector(sel); }
function num(n) { return Number(n).toLocaleString(); }

// ── LOGIN PAGE ────────────────────────────────────

const loginForm = qs('#login-form');
if (loginForm) {
  let pendingTicketId = null;

  const emailInput   = qs('#email');
  const passwordInput = qs('#password');
  const twofaGroup   = qs('#twofa-group');
  const twofaInput   = qs('#twofa-code');
  const submitBtn    = qs('#submit-btn');
  const errorMsg     = qs('#error-msg');

  function setError(msg) { errorMsg.textContent = msg; }
  function setLoading(on) {
    submitBtn.disabled = on;
    submitBtn.textContent = on ? 'Please wait…' : (pendingTicketId ? 'Verify Code' : 'Sign In');
  }

  loginForm.addEventListener('submit', async (e) => {
    e.preventDefault();
    setError('');
    setLoading(true);

    try {
      if (pendingTicketId) {
        // 2FA step
        const code = twofaInput.value.trim();
        if (!code) { setError('Enter the code from your email/authenticator.'); setLoading(false); return; }

        const res = await fetch('/api/verify-2fa', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ code, ticketId: pendingTicketId }),
        });
        const data = await res.json();
        if (!res.ok) { setError(data.error ?? 'Verification failed.'); setLoading(false); return; }

        sessionStorage.setItem('r6player', JSON.stringify(data.playerData));
        window.location.href = 'locker.html';

      } else {
        // Initial login
        const email    = emailInput.value.trim();
        const password = passwordInput.value;

        const res = await fetch('/api/login', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ email, password }),
        });
        const data = await res.json();
        if (!res.ok) { setError(data.error ?? 'Login failed.'); setLoading(false); return; }

        if (data.requires2FA) {
          pendingTicketId = data.ticketId;
          twofaGroup.classList.add('visible');
          twofaInput.focus();
          submitBtn.textContent = 'Verify Code';
          setLoading(false);
          return;
        }

        sessionStorage.setItem('r6player', JSON.stringify(data.playerData));
        window.location.href = 'locker.html';
      }
    } catch {
      setError('Network error. Is the server running?');
      setLoading(false);
    }
  });
}

// ── LOCKER PAGE ───────────────────────────────────

const lockerPage = qs('.locker-page');
if (lockerPage) {
  const raw = sessionStorage.getItem('r6player');
  if (!raw) { window.location.href = 'index.html'; }

  const player = JSON.parse(raw);
  renderLocker(player);
}

function renderLocker(p) {
  // Top bar
  const avatar = qs('#avatar');
  avatar.src = p.avatar;
  avatar.onerror = () => { avatar.src = 'https://ubisoft-avatars.akamaized.net/default/default_146_146.png'; };
  qs('#username').textContent = p.username;
  qs('#level').textContent = p.level;
  qs('#renown').textContent = num(p.renown);
  qs('#credits').textContent = num(p.credits);

  // Linked platforms
  const platformsEl = qs('#platforms');
  const platformLabels = { uplay: 'Ubisoft', psn: 'PlayStation', xbl: 'Xbox', steam: 'Steam' };
  (p.linkedPlatforms ?? []).forEach(plat => {
    const badge = document.createElement('span');
    badge.className = 'platform-badge';
    badge.textContent = platformLabels[plat] ?? plat;
    platformsEl.appendChild(badge);
  });

  // Season ranks
  const rankCards = qs('#rank-cards');
  if (!p.seasonRanks?.length) {
    rankCards.innerHTML = '<div style="color:var(--text-muted);font-size:.8rem;">No ranked seasons found</div>';
  } else {
    p.seasonRanks.forEach(r => {
      const card = document.createElement('div');
      card.className = `rank-card tier-${r.rankTier}`;
      const initials = r.rankName.split(' ').map(w => w[0]).join('').slice(0, 2);
      card.innerHTML = `
        <div class="rank-icon">${initials}</div>
        <div class="rank-info">
          <div class="rank-season-name">${r.seasonName}</div>
          <div class="rank-name-label">${r.rankName}</div>
        </div>
      `;
      rankCards.appendChild(card);
    });
  }

  // Inventory sections
  renderSection('seasonals',     p.seasonals     ?? []);
  renderSection('universals',    p.universals    ?? []);
  renderSection('blackices',     p.blackIces     ?? []);
  renderSection('rankedcharms',  p.rankedCharms  ?? []);
  renderSection('attachmentskins', p.attachmentSkins ?? []);
}

function renderSection(key, items) {
  qs(`#${key}-count`).textContent = items.length;
  const container = qs(`#${key}-pills`);

  if (!items.length) {
    const empty = document.createElement('span');
    empty.className = 'pills-empty';
    empty.textContent = 'None';
    container.appendChild(empty);
    return;
  }

  items.forEach(item => {
    const pill = document.createElement('button');
    pill.className = 'pill';
    pill.textContent = item.name;
    pill.addEventListener('click', () => openModal(item));
    container.appendChild(pill);
  });
}

// ── MODAL ─────────────────────────────────────────

const backdrop = qs('#modal-backdrop');
if (backdrop) {
  qs('#modal-close').addEventListener('click', closeModal);
  backdrop.addEventListener('click', (e) => { if (e.target === backdrop) closeModal(); });
}

function openModal(item) {
  const img = qs('#modal-image');
  img.src = item.image ?? '';
  if (!item.image) img.src = 'data:image/svg+xml,<svg xmlns="http://www.w3.org/2000/svg" width="180" height="180"><rect width="180" height="180" fill="%231c1c1c"/><text x="50%25" y="50%25" dominant-baseline="middle" text-anchor="middle" fill="%23555" font-size="40">?</text></svg>';

  qs('#modal-name').textContent = item.name;

  const rarityEl = qs('#modal-rarity');
  const rarityKey = (item.rarity ?? 'standard').toLowerCase();
  rarityEl.textContent = item.rarity ?? 'Standard';
  rarityEl.className = `badge badge-rarity-${rarityKey}`;

  qs('#modal-type').textContent = item.type ?? 'Item';

  backdrop.classList.add('open');
}

function closeModal() {
  backdrop.classList.remove('open');
}
